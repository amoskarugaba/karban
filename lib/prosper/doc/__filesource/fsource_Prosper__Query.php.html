  360.   /**
  361.    * Factory function that creates a select statement query object.
  362.    * @param varargs [optional] If nothing is provided assumes 'select *'
  363.    *       otherwise processes each argument as a column,
  364.    *        array arguments are processed as any number of aliased columns where column_name => alias
  365.    * @return Query Select Statement Query Object
  366.    */
  367.   static function select() {
  368.     $query = new Query('select ', self::SELECT_STMT);
  369.     
  370.     if(func_num_args() == 0) {
  371.       $columns = "*";
  372.     } else {
  373.       $args = func_get_args();
  374.       foreach($args as $arg) {
  375.         if(is_array($arg)) {
  376.           foreach($arg as $key => $value) {
  377.             $parts[] = $query->quote($key) . $query->alias($value);
  378.           }
  379.         } else {
  380.           $parts[] = $query->quote($arg);
  381.         }
  382.       }
  383.       $columns = implode(', ', $parts);
  384.     }
  385.     $query->sql .= $columns;
  386.     return $query;
  387.   }

  516.   /**
  517.    * Used by Query::where, Query::on, and Query::having to parse conditional strings,
  518.    * interpolating parameters, quoting, escaping, and cross-platform
  519.    * replacements where needed
  520.    * @param string $predicate result predicate, i.e. 'where' or 'on'
  521.    * @param string $clause conditional clause
  522.    * @see Query::where($clause)
  523.    * @see Query::on($clause)
  524.    * @see Query::having($clause)
  525.    */
  526.   function conditional($predicate, $clause, $args) {
  527.     $result = ' ' . $predicate;
  528.     $named = $args[count($args) - 1];    
  529.     while($token = Token::next($clause)) {
  530.       $result .= $this->process_token($clause, $token, $args, $named);
  531.     }
  532.     $this->sql .= $result;
  533.     return $this;
  534.   }

  536.   /**
  537.    * Processes a token
  538.    * @param string $clause sql cluase to process, by reference
  539.    * @param array $token token array
  540.    * @param array $args arguments used for interpolation, by reference
  541.    * @param array $named associative array
  542.    * @return string processed clause
  543.    */
  544.   function process_token(&$clause, $token, &$args, $named) {
  545.     switch($token['type']) {
  546.       case Token::BOOLEAN:
  547.         return ' ' . (strtolower($token['token']) == 'true' ? $this->true_value() : $this->false_value());
  548.         break;
  549.       case Token::LITERAL: 
  550.         return ' ' . $this->escape($token['token']);
  551.         break;
  552.       case Token::MODULUS:
  553.         $func_args = $this->parse_func_args($clause, $args, $named);
  554.         if(count($func_args) == 2) {
  555.           return $this->modulus($func_args[0], $func_args[1]);
  556.         }
  557.         break;
  558.       case Token::NAMED_PARAM:
  559.         return $this->parameterize($named[substr($token['token'], 1)]);
  560.         break;
  561.       case Token::PARAMETER:
  562.         return $this->parameterize(array_shift($args));
  563.         break;
  564.       case Token::SQL_ENTITY:
  565.         return ' ' . $this->quote($token['token']);
  566.         break;
  567.       default:
  568.         return ' ' . $token['token'];
  569.         break;
  570.     }
  571.   }

  573.   /**
  574.    * Performs safe parameter interpolation
  575.    * @param mixed $value Value to interpolate, can be an array (will be imploded with commas), a subquery, or a literal value.
  576.    * @return string parameterized fragment
  577.    */
  578.   function parameterize($value) {
  579.     if(is_array($value)) {
  580.       return ' ' . implode(', ', array_map(array($this, 'escape'), $value));
  581.     } else if($value instanceof Query) {
  582.       return ' ' . $value;
  583.     } else {
  584.       return ' ' . $this->escape($value);
  585.     }
  586.   }

  588.   /**
  589.    * Parses sql function arguments into a php array
  590.    * @param string $clause sql clause to process, by reference
  591.    * @param array $args arguments used for interpolation, by reference
  592.    * @param array $named named interpolation arguments
  593.    * @return array arguments
  594.    */
  595.   function parse_func_args(&$clause, &$args, $named) {
  596.     $token = Token::next($clause);    
  597.     if($token['type'] == Token::OPEN_PAREN) {
  598.       $depth = 1;
  599.       $temp = '';
  600.       while($depth > 0) {
  601.         $token = Token::next($clause);
  602.         switch($token['type']) {
  603.           case Token::OPEN_PAREN:
  604.             $depth += 1;
  605.             $temp .= $this->process_token($clause, $token, $args, $named);
  606.             break;
  607.           case Token::CLOSE_PAREN:
  608.             $depth -= 1;
  609.             if($depth == 0) {
  610.               $result[] = $temp;
  611.             }
  612.             $temp .= $this->process_token($clause, $token, $args, $named);
  613.             break;
  614.           case Token::COMMA:
  615.             if($depth == 1) {
  616.               $result[] = $temp;
  617.               $temp = '';
  618.             }
  619.             break;
  620.           default:
  621.             $temp .= $this->process_token($clause, $token, $args, $named);
  622.             break;
  623.         }
  624.       }
  625.       return $result;
  626.     } else {
  627.       //Something is wrong -- return the processed token
  628.       return $this->process_token($clause, $tkn, $args, $named);
  629.     }
  630.   }

  684.    * This function takes an optional variadic argument for filtering the values
  685.    *   Example:
  686.    *     Assume we have a $_POST array where print_r($_POST) =>
  687.    *      array ( 'first_name' => 'Matt',

  691.    *       is identical to
  692.    *     ->values(array('first_name' => $_POST['first_name'], 'last_name' => $_POST['last_name']))
  693.    * @param varargs $filter [optional] list of elements to pull out of $values
  694.    * @param array $values Array of values with column_name => insert_value

  696.    */
  697.   function values($values) {

  699.       $filters = func_get_args();
  700.       $values = array_pop($filters);      //Remove $values
  701.       foreach($filters as $filter) {
  702.         if(array_key_exists($filter, $values)) {
  703.           $cols[] = $this->quote($filter);
  704.           $vals[] = $this->escape($values[$filter]);
  705.         }
  706.       }
  707.     } else {
  708.       foreach($values as $key => $value) {
  709.         $cols[] = $this->quote($key);
  710.         $vals[] = $this->escape($value);
  711.       }
  712.     }
  713.     
  714.     if(is_array($cols) && is_array($vals)) {
  715.       $this->sql .= ' (' . implode(', ', $cols) . ') values (' . implode(', ', $vals) . ')';
  716.     }
  717.     return $this;
  718.   }
